% Custom commands

% define headings for workplaces, e.g. caltech
\newcommand{\place}[2]{
  \vspace{-2pt}
  \item
    \begin{tabular*}
      {0.97\textwidth}[t] % tabular env, top aligned
      {l@{\extracolsep{\fill}}r} % left & right cols + expanded padding
      \textbf{#1} % employer/university 
      & \textit{\footnotesize{#2}} \\  % location
    \end{tabular*}
    \vspace{-7pt}
}

% define subsections for roles, e.g. grad student
\newcommand{\role}[2]{
  \vspace{-2pt}
  \item
    \begin{tabular*}
      {0.97\textwidth}[t] % tabular env, top aligned
      {l@{\extracolsep{\fill}}r} % left & right cols + expanded padding
      {\small #1}  % role
      & {\footnotesize{#2}} \\  % dates held
    \end{tabular*}
    \vspace{-7pt}
}

% variation for things like courses taught, projects led
\newcommand{\entry}[3]{
  \vspace{-2pt}
  \item
    \begin{tabular*}
      {0.97\textwidth} % tabular env, top aligned
      {l@{\extracolsep{\fill}}r} % left & right cols + expanded padding
      {\small \textbf{#1} #2}  % class/org, role, institution
      & {\footnotesize{#3}} \\  % dates held
    \end{tabular*}
    \vspace{-4pt}
}

% publications

  \item
  % \begin{tabular*}{0.97\textwidth}[t]{@{}p{0.97\textwidth}@{}}

    \small#1 % authors
    \small``#2'' % title
    \small\textbf{\emph{#3}} % journal
    \small(#4). % year
    \small#5 % doi/link
  \end{tabular*}
  \vspace{-2pt}
}
